        // Render Key Moments

        function renderMoments() {

            momentsList.innerHTML = '';

            moments.forEach((scene, sceneIdx) => {

                const sceneMarker = document.createElement('div');

                sceneMarker.className = 'scene-marker';

                sceneMarker.textContent = scene.scene;

                momentsList.appendChild(sceneMarker);


                scene.items.forEach((moment, idx) => {

                    const momentItem = document.createElement('div');

                    momentItem.className = 'moment-item';

                    momentItem.dataset.time = moment.time;

                    

                    const location = document.createElement('div');

                    location.className = 'moment-location';

                    if (currentTab === 'nofear') {

                        location.textContent = `No Fear: Page ${moment.noFearPage} | Original: Page ${moment.originalPage} | Lines ${moment.lines}`;

                    } else {

                        location.textContent = `Page ${moment.originalPage} | Lines ${moment.lines}`;

                    }

                    

                    const description = document.createElement('div');

                    description.className = 'moment-description';

                    description.textContent = moment.description;

                    

                    momentItem.appendChild(location);

                    momentItem.appendChild(description);

                    momentItem.addEventListener('click', () => {

                        audio.currentTime = moment.time;

                        if (audio.paused) {

                            audio.play();

                            playPauseBtn.textContent = '⏸ Pause';

                        }

                    });

                    

                    momentsList.appendChild(momentItem);

                });

            });

        }


        // Update active moment

        function updateActiveMoment() {

            const currentTime = audio.currentTime;

            const allMoments = [];

            moments.forEach(scene => {

                scene.items.forEach(item => allMoments.push(item));

            });


            let activeIdx = -1;

            for (let i = allMoments.length - 1; i >= 0; i--) {

                if (currentTime >= allMoments[i].time) {

                    activeIdx = i;

                    break;

                }

            }


            if (activeIdx !== currentMomentIndex) {

                currentMomentIndex = activeIdx;

                const momentItems = document.querySelectorAll('.moment-item');

                momentItems.forEach((item, idx) => {

                    item.classList.toggle('active', idx === activeIdx);

                });


                if (activeIdx >= 0) {

                    const activeMoment = allMoments[activeIdx];

                    const pageNum = currentTab === 'nofear' ? activeMoment.noFearPage : activeMoment.originalPage;

                    pdfFrame.src = pdfFrame.src.split('#')[0] + `#page=${pageNum}`;

                    

                    // Scroll to active moment

                    const activeElement = momentItems[activeIdx];

                    if (activeElement) {

                        activeElement.scrollIntoView({ behavior: 'smooth', block: 'center' });

                    }

                }

            }

        }


        // Audio Controls

            } else {
